### Describe the bug

Kernels created by the `cudaq::kernel_builder` class can't be translated to OpenQASM by `cudaq-translate`. This is because they don't contain any function with the `cudaq-entrypoint` attribute.

### Steps to reproduce the bug

Compile the following code using `nvq++ kernel.cpp`:
```c++
#include <cudaq.h>

#include <iostream>

int main() {
        auto kernel = cudaq::make_kernel();
        auto qubit = kernel.qalloc(1);
        kernel.x(qubit[0]);
        kernel.mz(qubit[0]);
        std::cout << kernel.to_quake() << std::endl;
        return 0;
}
```

The result is the following MLIR code:
```
module {
  func.func @__nvqpp__mlirgen____nvqppBuilderKernel_367535629127() {
    %0 = quake.alloca !quake.veq<1>
    %1 = quake.extract_ref %0[0] : (!quake.veq<1>) -> !quake.ref
    quake.x %1 : (!quake.ref) -> ()
    %2 = quake.mz %1 : (!quake.ref) -> i1 {registerName = ""}
    return
  }
}
```

When the file is translated using `cudaq-translate --convert-to=openqasm kernel.qke`, the output is
```
// Code generated by NVIDIA's nvq++ compiler
OPENQASM 2.0;

include "qelib1.inc";

[quake-translate] Dumping Module after error.
[quake-translate] Reported Error: see current operation:
"func.func"() ({
  %0 = "quake.alloca"() : () -> !quake.veq<1>
  %1 = "quake.extract_ref"(%0) {rawIndex = 0 : i64} : (!quake.veq<1>) -> !quake.ref
  "quake.x"(%1) {operand_segment_sizes = array<i32: 0, 0, 1>} : (!quake.ref) -> ()
  %2 = "quake.mz"(%1) {registerName = ""} : (!quake.ref) -> i1
  "func.return"() : () -> ()
}) {function_type = () -> (), sym_name = "__nvqpp__mlirgen____nvqppBuilderKernel_367535629127"} : () -> ()
loc("kernel.qke":2:3): error: cannot translated classical functions
[quake-translate] Dumping Module after error.
loc("kernel.qke":1:1): error: translation failed
LLVM ERROR: fatal error, aborting.
Aborted
```

### Expected behavior

The output of the C++ program should be:
```
module {
  func.func @__nvqpp__mlirgen____nvqppBuilderKernel_367535629127() attributes {"cudaq-entrypoint"} {
    %0 = quake.alloca !quake.veq<1>
    %1 = quake.extract_ref %0[0] : (!quake.veq<1>) -> !quake.ref
    quake.x %1 : (!quake.ref) -> ()
    %2 = quake.mz %1 : (!quake.ref) -> i1 {registerName = ""}
    return
  }
}
```

Which will then be translated to:
```
// Code generated by NVIDIA's nvq++ compiler
OPENQASM 2.0;

include "qelib1.inc";

qreg var0[1];
x var0[0];
creg var2[1];
measure var0[0] -> var2[0];
```
