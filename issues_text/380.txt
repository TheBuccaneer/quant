### Describe the bug

Quake code in which a kernel is called by another kernel can't be translated to OpenQASM.
This is due to two reasons:
- The CallOp instruction isn't supported by the translator.
- The resulting gate definition's name starts with an underscore (`_`), which isn't allowed in OpenQASM.

### Steps to reproduce the bug

Compile the following code using `nvq++ kernel.ccp`:
```c++
#include <cudaq.h>

#include <numbers>
#include <iostream>

int main() {
        auto [kernel1, qubit1] = cudaq::make_kernel<cudaq::qubit>();
        kernel1.x(qubit1);

        auto kernel2 = cudaq::make_kernel();
        auto qubit = kernel2.qalloc();
        kernel2.call(kernel1, qubit);
        kernel2.mz(qubit);

        std::cout << kernel2.to_quake() << std::endl;
        return 0;
}
```

The result is the following MLIR code:
```
module {
  func.func @__nvqpp__mlirgen____nvqppBuilderKernel_093606261879() {
    %0 = quake.alloca !quake.ref
    call @__nvqpp__mlirgen____nvqppBuilderKernel_367535629127(%0) : (!quake.ref) -> ()
    %1 = quake.mz %0 : (!quake.ref) -> i1 {registerName = ""}
    return
  }
  func.func @__nvqpp__mlirgen____nvqppBuilderKernel_367535629127(%arg0: !quake.ref) {
    quake.x %arg0 : (!quake.ref) -> ()
    return
  }
}
```

When the file is translated using `cudaq-translate --convert-to=openqasm kernel.qke`, the output is:
```
// Code generated by NVIDIA's nvq++ compiler
OPENQASM 2.0;

include "qelib1.inc";

// Code generated by NVIDIA's nvq++ compiler
OPENQASM 2.0;

include "qelib1.inc";

gate __nvqpp__mlirgen____nvqppBuilderKernel_367535629127 q0 {
  x q0;
}

qreg var0[1];
[quake-translate] Dumping Module after error.
[quake-translate] Reported Error: see current operation: "func.call"(%0) {callee = @__nvqpp__mlirgen____nvqppBuilderKernel_367535629127} : (!quake.ref) -> ()
loc("kernel.qke":4:5): error: 'func.call' op unable to translate op to OpenQASM 2.0
[quake-translate] Dumping Module after error.
loc("kernel.qke":0:0): error: translation failed
LLVM ERROR: fatal error, aborting.
Aborted
```


### Expected behavior

The MLIR code should be successfully translated to:
```
// Code generated by NVIDIA's nvq++ compiler
OPENQASM 2.0;

include "qelib1.inc";
gate x__nvqpp__mlirgen____nvqppBuilderKernel_367535629127 q0 {
  x q0;
}

qreg var0[1];
x__nvqpp__mlirgen____nvqppBuilderKernel_367535629127 var0[0];
creg var1[1];
measure var0[0] -> var1[0];
```

Or to an equivalent OpenQASM code.
