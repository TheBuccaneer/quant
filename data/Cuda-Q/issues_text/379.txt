### Describe the bug

When translating Quake to OpenQASM, `r1` gates are translated to `cu1` gates instead of `u1` gates.

### Steps to reproduce the bug

Compile the following code with `nvq++ kernel.cpp`:
```c++
#include <cudaq.h>

#include <numbers>
#include <iostream>

int main() {
        auto kernel = cudaq::make_kernel();
        auto qubit = kernel.qalloc(1);
        kernel.r1(std::numbers::pi_v<double>, qubit[0]);
        kernel.mz(qubit[0]);
        std::string to_replace = "367535629127()";
        std::string replacement = "367535629127() attributes {\"cudaq-entrypoint\"}";
        std::string quake = kernel.to_quake();
        std::cout << quake.replace(quake.find(to_replace), to_replace.size(), replacement) << std::endl;
        return 0;
}
```

The result is the following MLIR code:
```
module {
  func.func @__nvqpp__mlirgen____nvqppBuilderKernel_367535629127() attributes {"cudaq-entrypoint"} {
    %cst = arith.constant 3.1415926535897931 : f64
    %0 = quake.alloca !quake.veq<1>
    %1 = quake.extract_ref %0[0] : (!quake.veq<1>) -> !quake.ref
    quake.r1 (%cst) %1 : (f64, !quake.ref) -> ()
    %2 = quake.mz %1 : (!quake.ref) -> i1 {registerName = ""}
    return
  }
}
```

Translate the MLIR code to OpenQASM with `cudaq-translate --convert-to=openqasm kernel.qke`. The output is:
```
// Code generated by NVIDIA's nvq++ compiler
OPENQASM 2.0;

include "qelib1.inc";

qreg var0[1];
cu1(3.141593e+00) var0[0];
creg var2[1];
measure var0[0] -> var2[0];
```

### Expected behavior

The resulting OpenQASM code should be:
```
// Code generated by NVIDIA's nvq++ compiler
OPENQASM 2.0;

include "qelib1.inc";

qreg var0[1];
u1(3.141593e+00) var0[0];
creg var2[1];
measure var0[0] -> var2[0];
```
