### Informations
- **Qiskit version**: 1.3.1
- **Qiskit Aer version**: 0.15.1
- **Python version**: 3.11.11

### What is the current behavior?

When the transpiler processes a complex unitary gate, the statevectors before and after transpilation differ significantly.



### Steps to reproduce the problem
Please see the sample code and output below.
```python
from qiskit import QuantumCircuit, transpile
from qiskit.quantum_info import Statevector
from qiskit_aer import AerSimulator
import numpy as np


def compare_statevectors(vec1, vec2, tol=1e-10):
    """Compares two statevectors and prints significant differences."""
    differences = np.abs(vec1 - vec2)
    max_diff = np.max(differences)
    print(f"Max difference: {max_diff}")
    if max_diff > tol:
        print("Significant differences found:")
        for i, diff in enumerate(differences):
            if diff > tol:
                print(f"Index {i}: {vec1[i]} vs {vec2[i]} (diff={diff})")
    else:
        print("Statevectors are effectively identical.")


circuit_qasm = """

include "qelib1.inc";

gate circuit_879 q0,q1 { u(pi/2,-pi/2,-pi) q0; u(1.6719020266110614,-pi/2,-pi) q1; cx q0,q1; u(pi/2,0,pi/2) q0; u(1.4702087094713374,1.4691752037207921,-3.1313531404558756) q1; cx q0,q1; u(pi/2,pi/2,-pi) q0; u(0.10110569981616473,0,-pi/2) q1; }
gate circuit_882 q0,q1 { u(0,-pi,-pi) q0; u(pi/2,-pi,-0.10110569981616457) q1; cx q0,q1; u(pi,1.3258176636680323,1.3258176636680323) q0; u(pi/2,3.0404869537736285,-pi) q1; }
gate circuit_885 q0,q1 { u(0,-pi,-pi) q0; u(0,-pi/2,pi/2) q1; cx q0,q1; u(0,-2.297438667476622,2.297438667476622) q0; u(0,-0.024692629465313587,0.024692629465313587) q1; }
gate unitary q0,q1,q2 { circuit_876 q0,q1; u(0,0,-pi/4) q2; cx q0,q2; u(0,0,-pi/4) q2; cx q1,q2; u(0,0,-pi/4) q2; cx q0,q2; u(0,0,-pi/4) q2; cx q1,q2; circuit_879 q0,q1; u(pi/8,0,0) q2; u(pi/2,0,pi) q2; cx q0,q2; u(pi/2,0,pi) q2; u(-pi/8,0,0) q2; u(pi/2,0,pi) q2; cx q1,q2; u(pi/2,0,pi) q2; u(pi/8,0,0) q2; u(pi/2,0,pi) q2; cx q0,q2; circuit_882 q0,q1; u(pi/2,0,pi) q2; u(-pi/8,0,0) q2; u(0,0,-pi/2) q2; cx q0,q2; cx q1,q2; cx q0,q2; u(0,0,-pi/2) q2; cx q1,q2; circuit_885 q0,q1; }

qreg q[3];
h q[2];
unitary q[0],q[1],q[2];
"""

qc = QuantumCircuit.from_qasm_str(circuit_qasm)
orig_statevector = Statevector(qc)

qc_transpiled = transpile(
    qc,
    AerSimulator(),
    basis_gates=["u1", "u2", "u3", "cx"],
    optimization_level=0,
)
qc_transpiled_statevector = Statevector(qc_transpiled)

if orig_statevector.equiv(qc_transpiled_statevector):
    print("All is well!")
else:
    print(f"Original circuit:\n{qc.draw()}")
    print("Statevectors are not equivalent!")
    print(f"Original statevector:\n{orig_statevector}")
    print(f"Transpiled statevector:\n{qc_transpiled_statevector}")
    compare_statevectors(orig_statevector, qc_transpiled_statevector)

```
